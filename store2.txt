        img.alt = card.dataset.name || 'produk';
        img.loading = 'lazy';
        wrap.appendChild(img);

        // Fallback: tampilkan emoji
        const fb = document.createElement('div');
        fb.className = 'product-img-fallback';
        fb.textContent = icon;
        wrap.appendChild(fb);
      }
    });
  }

  const GAME_REVIEW_MAP = {
    'adoptme':   ['Adopt Me!'],
    'tsunami':   ['Escape Tsunami', 'Escape Tsunami For Brainrot'],
    'bloxfruit': ['Blox Fruit'],
    'forest':    ['99 Night', '99 Night in the Forest'],

  function buildGameReviews() {
    Object.keys(GAME_REVIEW_MAP).forEach(gameId => {
      const keywords = GAME_REVIEW_MAP[gameId];
      const track = document.getElementById('reviews-' + gameId);
      if (!track) return;
      const matched = REVIEWS.filter(r =>
        keywords.some(k => r.game.toLowerCase().includes(k.toLowerCase()))
      );
      if (matched.length === 0) {
        track.parentElement.parentElement.innerHTML =
          '<div class="no-review-note">Belum ada ulasan untuk game ini. Jadilah yang pertama! ✍️</div>';
        return;
      }
      const list = matched.length < 3 ? [...matched, ...matched, ...matched] : [...matched, ...matched];
      track.innerHTML = list.map(r => {
        const stars = '★'.repeat(r.stars) + '☆'.repeat(5 - r.stars);
        const initials = r.name.split(' ').map(w => w[0]).join('').slice(0,2).toUpperCase();
        return `
          <div class="game-review-card">
            <div class="review-stars">${stars}</div>
            <div class="review-text">"${r.text}"</div>
            <div class="review-author">
              <div class="review-avatar">${initials}</div>
              <div class="review-name">${r.name}</div>
            </div>
          </div>`;
      }).join('');
      track.style.animationDuration = (list.length * 4) + 's';
    });
  }
  buildGameReviews();

  // ---- PROMO SLIDER ----
  const TOTAL_SLIDES = 4;
  let promoIndex = 0;
  let promoTimer;
  function buildPromoDots() {
    const dots = document.getElementById('promoDots');
    dots.innerHTML = Array.from({length: TOTAL_SLIDES}, (_, i) =>
      `<div class="promo-dot ${i===0?'active':''}" onclick="goPromo(${i})"></div>`
    ).join('');
  }
  function goPromo(idx) {
    promoIndex = idx;
    document.getElementById('promoTrack').style.transform = `translateX(-${idx * 100}%)`;
    document.querySelectorAll('.promo-dot').forEach((d, i) => d.classList.toggle('active', i === idx));
    clearInterval(promoTimer);
    promoTimer = setInterval(nextPromo, 3500);
  }
  function nextPromo() { goPromo((promoIndex + 1) % TOTAL_SLIDES); }
  buildPromoDots();
  promoTimer = setInterval(nextPromo, 3500);
  (function(){
    let startX = 0;
    const el = document.getElementById('promoTrack');
    el.addEventListener('touchstart', e => { startX = e.touches[0].clientX; }, {passive:true});
    el.addEventListener('touchend', e => {
      const diff = startX - e.changedTouches[0].clientX;
      if (Math.abs(diff) > 40) goPromo(diff > 0
        ? (promoIndex + 1) % TOTAL_SLIDES
        : (promoIndex - 1 + TOTAL_SLIDES) % TOTAL_SLIDES);
    });
  })();

  // ---- LANG DROPDOWN ----
  let currentLang = 'id';
  function toggleLangMenu() { document.getElementById('langMenu').classList.toggle('open'); }
  document.addEventListener('click', function(e) {
    if (!document.getElementById('langWrap').contains(e.target))
      document.getElementById('langMenu').classList.remove('open');
  });
  const LANG_LABELS = { id: '🇮🇩 ID', en: '🇬🇧 EN', my: '🇲🇾 MY' };

  function formatPrice(idr, lang) {
    if (lang === 'en') {
      const usd = idr / CONFIG.rates.USD;
      const rounded = Math.ceil(usd / 0.05) * 0.05;
      return '$' + rounded.toFixed(2);
    } else if (lang === 'my') {
      const myr = idr / CONFIG.rates.MYR;
      const rounded = Math.ceil(myr / 0.5) * 0.5;
      return 'RM ' + rounded.toFixed(2);
    } else {
      return 'Rp ' + parseInt(idr).toLocaleString('id-ID');
    }
  }
  function updatePrices(lang) {
    document.querySelectorAll('.product-card').forEach(card => {
      const idr    = parseInt(card.dataset.price);
      const oldIdr = parseInt(card.dataset.oldPrice);
      const priceEl = card.querySelector('.product-price');
      const oldEl   = card.querySelector('.product-old-price');
      if (priceEl) priceEl.textContent = formatPrice(idr, lang);
      if (oldEl && oldIdr) oldEl.textContent = formatPrice(oldIdr, lang);
    });
    const modalPriceEl = document.getElementById('buyNowProductPrice');
    if (modalPriceEl && modalPriceEl.dataset.idr) {
      modalPriceEl.textContent = formatPrice(parseInt(modalPriceEl.dataset.idr), lang);
    }
  }
  function setLang(lang) {
    currentLang = lang;
    const t = LANG[lang];
    document.getElementById('langMenu').classList.remove('open');
    document.getElementById('langBtn').innerHTML = `${LANG_LABELS[lang]} <i class="bi bi-chevron-down" style="font-size:.6rem"></i>`;
    document.querySelectorAll('.lang-menu-item').forEach(item => {
      item.classList.toggle('active', item.dataset.lang === lang);
    });
    document.querySelectorAll('[data-i18n]').forEach(el => {
      const key = el.getAttribute('data-i18n');
      if (t[key] !== undefined) el.innerHTML = t[key];
    });
    const isDark = document.documentElement.getAttribute('data-theme') === 'dark';
    document.getElementById('themeLabel').textContent = isDark ? t.themeTerang : t.themeGelap;
    document.querySelectorAll('.btn-buy-wa').forEach(b => {
      b.innerHTML = `<i class="bi bi-whatsapp"></i> ${t.beliWA}`;
    });
    updatePrices(lang);
  }
  function openWAChat() {
    const t = LANG[currentLang];
    window.open(`https://wa.me/${CONFIG.waNumber}?text=${encodeURIComponent(t.waMsg(CONFIG.storeName))}`, '_blank');
  }
